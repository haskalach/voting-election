### 2. Services & Dependency Injection

âś… **GOOD - Singleton services in core module**:

```typescript
// auth.service.ts
@Injectable({ providedIn: 'root' })
export class AuthService {
  private _accessToken = signal<string | null>(null);
  private readonly REFRESH_KEY = 'refresh_token';

  constructor(private http: HttpClient) {}

  login(email: string, password: string): Observable<LoginResponse> {
    return this.http.post<LoginResponse>('/api/auth/login', { email, password })
      .pipe(
        tap(response => {
          this._accessToken.set(response.accessToken);
          localStorage.setItem(this.REFRESH_KEY, response.refreshToken);
        })
      );
  }

  getToken(): string | null {
    return this._accessToken();
  }
}

// In components:
constructor(private authService: AuthService) {}
```

âťŚ **BAD - Services in component providers or multiple providers**:

```typescript
// âťŚ Created new instance per component
@Component({
  selector: "app-login",
  providers: [AuthService], // âťŚ New instance per component
})
export class LoginComponent {}

// âťŚ Service instantiation in component
export class LoginComponent {
  private authService = new AuthService(); // âťŚ Manual instantiation
}
```

### 3. Reactive Forms & Validation

âś… **GOOD - Reactive forms with validators**:

```typescript
export class LoginComponent {
  loginForm = new FormGroup({
    email: new FormControl('', [Validators.required, Validators.email]),
    password: new FormControl('', [Validators.required, Validators.minLength(8)])
  });

  onSubmit() {
    if (this.loginForm.valid) {
      const { email, password } = this.loginForm.value;
      this.authService.login(email, password).subscribe({
        next: (response) => console.log('Login successful'),
        error: (error) => this.onLoginError(error)
      });
    }
  }
}

// Template:

  <div *ngIf="loginForm.get('email')?.hasError('email')">
    Invalid email format
  </div>
  <button [disabled]="loginForm.invalid">Login</button>
</form>
```

âťŚ **BAD - Template-driven forms (old pattern)**:


### 4. RxJS Patterns - Signals & Observables

âś… **GOOD - Signals for state, observables for streams**:

```typescript
// State as signals (in-memory, reactive)
@Injectable({ providedIn: 'root' })
export class OrganizationService {
  private organizationSubject = new BehaviorSubject<Organization | null>(null);
  organizations$ = this.organizationSubject.asObservable();

  // Or using angular signals
  selectedOrganization = signal<Organization | null>(null);

  loadOrganizations(): Observable<Organization[]> {
    return this.http.get<Organization[]>('/api/organizations')
      .pipe(
        tap(orgs => this.organizations$.next(orgs))
      );
  }
}

// In component:
export class OrganizationListComponent {
  organizations$ = this.orgService.organizations$;

  constructor(private orgService: OrganizationService) {}

  ngOnInit() {
    this.orgService.loadOrganizations().subscribe();
  }
}

// Template uses async pipe:
<div *ngFor="let org of organizations$ | async">
  {{ org.name }}
</div>
```

âťŚ **BAD - No subscription management**:

```typescript
// âťŚ Memory leaks from unsubscribed observables
export class OrganizationListComponent implements OnInit {
  organizations: Organization[] = [];

  ngOnInit() {
    this.orgService.organizations$.subscribe((orgs) => {
      this.organizations = orgs; // âťŚ Manual subscription not unsubscribed
    });
  }
}
```

### 5. HTTP Interceptors

âś… **GOOD - JWT token injection + 401 refresh**:

```typescript
@Injectable()
export class JwtInterceptor implements HttpInterceptor {
  constructor(private authService: AuthService) {}

  intercept(
    req: HttpRequest<unknown>,
    next: HttpHandler,
  ): Observable<HttpEvent<unknown>> {
    const token = this.authService.getToken();

    if (token) {
      req = req.clone({
        setHeaders: {
          Authorization: `Bearer ${token}`,
        },
      });
    }

    return next.handle(req).pipe(
      catchError((error: HttpErrorResponse) => {
        if (error.status === 401) {
          return this.authService.refreshToken().pipe(
            switchMap((response) => {
              const newReq = req.clone({
                setHeaders: { Authorization: `Bearer ${response.accessToken}` },
              });
              return next.handle(newReq);
            }),
            catchError(() => this.authService.logout()),
          );
        }
        return throwError(() => error);
      }),
    );
  }
}
```

âťŚ **BAD - Inline token handling**:

```typescript
// âťŚ Token handling scattered in components
export class OrganizationListComponent {
  constructor(private http: HttpClient) {}

  loadOrgs() {
    const token = localStorage.getItem("token"); // âťŚ Direct LS access
    const headers = new HttpHeaders({ Authorization: `Bearer ${token}` });
    this.http.get<any>("/api/organizations", { headers }).subscribe();
  }
}
```

### 7. Error Handling

âś… **GOOD - Global error interceptor**:

```typescript
@Injectable()
export class ErrorInterceptor implements HttpInterceptor {
  intercept(
    req: HttpRequest<unknown>,
    next: HttpHandler,
  ): Observable<HttpEvent<unknown>> {
    return next.handle(req).pipe(
      catchError((error: HttpErrorResponse) => {
        let message = "An error occurred";

        if (error.status === 400) {
          message = error.error.message || "Invalid request";
        } else if (error.status === 403) {
          message = "You do not have permission";
        } else if (error.status === 404) {
          message = "Resource not found";
        } else if (error.status === 500) {
          message = "Server error";
        }

        console.error(message, error);
        return throwError(() => new Error(message));
      }),
    );
  }
}

// Component handles errors:
this.orgService.loadOrganizations().subscribe({
  next: (orgs) => (this.organizations = orgs),
  error: (error) => this.showError(error.message),
});
```

âťŚ **BAD - No error handling**:

```typescript
// âťŚ Errors silently fail
this.http.get("/api/organizations").subscribe((orgs) => {
  this.organizations = orgs;
});
```

### 8. Routing & Guards

âś… **GOOD - Route guards for auth & roles**:

```typescript
@Injectable({ providedIn: "root" })
export class AuthGuard implements CanActivate {
  constructor(
    private authService: AuthService,
    private router: Router,
  ) {}

  canActivate(): boolean {
    if (this.authService.isAuthenticated()) {
      return true;
    }
    this.router.navigate(["/login"]);
    return false;
  }
}

@Injectable({ providedIn: "root" })
export class RoleGuard implements CanActivate {
  constructor(private authService: AuthService) {}

  canActivate(route: ActivatedRouteSnapshot): boolean {
    const requiredRole = route.data["role"];
    return this.authService.hasRole(requiredRole);
  }
}

// Routes:
const routes: Routes = [
  { path: "login", component: LoginComponent },
  {
    path: "organizations",
    component: OrganizationListComponent,
    canActivate: [AuthGuard, RoleGuard],
    data: { role: ["SystemOwner", "Manager"] },
  },
];
```

### 10. Naming Conventions

| Category         | Convention                        | Example                                       |
| ---------------- | --------------------------------- | --------------------------------------------- |
| **Components**   | Descriptive, ends with .component | `LoginComponent`, `OrganizationListComponent` |
| **Services**     | Ends with .service                | `AuthService`, `OrganizationService`          |
| **Guards**       | Ends with .guard                  | `AuthGuard`, `RoleGuard`                      |
| **Interceptors** | Ends with .interceptor            | `JwtInterceptor`, `ErrorInterceptor`          |
| **Modules**      | Ends with .module                 | `SharedModule`, `OrganizationsModule`         |
| **Properties**   | camelCase                         | `organizationId`, `firstName`                 |
| **Methods**      | camelCase, verb first             | `loadOrganizations()`, `validateForm()`       |
| **Observables**  | End with $                        | `organizations$`, `currentUser$`              |
| **Signals**      | camelCase, no $                   | `selectedOrg`, `isLoading`                    |
| **Constants**    | UPPER_CASE                        | `API_BASE_URL`, `MAX_RETRIES`                 |

## When Modifying Frontend Code

### Adding a New Feature Module

1. **Generate module structure**:

   ```bash
   ng generate module features/my-feature
   ng generate component features/my-feature/my-feature
   ng generate component features/my-feature/my-feature-detail
   ng generate service features/my-feature/my-feature.service
   ```

2. **Create service** in `my-feature.service.ts`:

   ```typescript
   @Injectable({ providedIn: "root" })
   export class MyFeatureService {
     private myFeatures$ = new BehaviorSubject<MyFeature[]>([]);

     constructor(private http: HttpClient) {}

     getAll(): Observable<MyFeature[]> {
       return this.http
         .get<MyFeature[]>("/api/my-features")
         .pipe(tap((items) => this.myFeatures$.next(items)));
     }
   }
   ```

3. **Create component** with reactive forms:

   ```typescript
   @Component({
     selector: "app-my-feature-list",
     template: "...",
   })
   export class MyFeatureListComponent implements OnInit {
     myFeatures$ = this.service.myFeatures$;

     constructor(private service: MyFeatureService) {}

     ngOnInit() {
       this.service.getAll().subscribe();
     }
   }
   ```

4. **Add routing**:

   ```typescript
   const routes: Routes = [
     { path: "", component: MyFeatureListComponent },
     { path: ":id", component: MyFeatureDetailComponent },
   ];
   ```

5. **Register in AppRoutingModule** with lazy loading

### Consuming Backend API

1. **Use ApiService wrapper** (not raw HttpClient):

   ```typescript
   constructor(private api: ApiService) {}

   this.api.get<Organization[]>('/organizations').subscribe();
   this.api.post<Organization>('/organizations', createDto).subscribe();
   this.api.put<Organization>('/organizations/1', updateDto).subscribe();
   this.api.delete('/organizations/1').subscribe();
   ```

2. **Always use error handling**:
   ```typescript
   this.service.loadOrganizations().subscribe({
     next: (orgs) => (this.organizations = orgs),
     error: (error) => this.showErrorMessage(error),
     complete: () => console.log("Load complete"),
   });
   ```

## Security Checklist

When reviewing frontend changes:

- [ ] Access tokens never stored in localStorage (signals only)
- [ ] Refresh tokens in httpOnly cookies (if available)
- [ ] JWT interceptor validates token before sending
- [ ] 401 response triggers refresh or logout
- [ ] Route guards check authentication before navigation
- [ ] Role-based access validation
- [ ] XSS prevention (Angular's built-in sanitization)
- [ ] CSRF tokens included in requests (if configured)
- [ ] No sensitive data in local storage
- [ ] API calls only to configured backend domain
